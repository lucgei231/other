Code Preview

                
                    // Example of a simple asynchronous function
                    function fetchData() {
                        return new Promise((resolve) => {
                            setTimeout(() => {
                                resolve("Data fetched!");
                            }, 2000);
                        });
                    }

                    fetchData().then(data => console.log(data));
