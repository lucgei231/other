Example: Closures

Closures allow a function to access variables from an enclosing scope even after it leaves the scope in which it was declared.


function makeCounter() {
    let count = 0;
    return function() {
        count++;
        return count;
    };
}

const counter = makeCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

Exercise:

Implement a function that returns a function which adds a specific number to its argument.


function addX(x) {
    return function(y) {
        return x + y;
    };
}

const add5 = addX(5);
console.log(add5(10)); // 15
